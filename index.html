private static int[] GetArrayFromNumber(int number)
{
    int digitCount = (int)Math.Log10(number) + 1;
    int[] numbers = new int[digitCount];
    for (int i = 0; i < digitCount; i++)
    {
        numbers[i] = number % 10;
        number /= 10;
    }
    return numbers;
}
